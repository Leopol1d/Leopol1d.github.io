222. 完全二叉树的节点个数

BFS

/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public int countNodes(TreeNode root) {
        if (root == null)
            return 0;
        Queue<TreeNode> queue = new LinkedList<>();
        queue.offer(root);
        int count = 0;
        while (!queue.isEmpty()) {
            TreeNode cur = queue.poll();
            ++count;
            if (cur.left != null) {
                queue.offer(cur.left);
            }
            if (cur.right != null) {
                queue.offer(cur.right);
            }
        }
        return count;
    }
}

416. 分割等和子集

0-1背包

class Solution {
    public boolean canPartition(int[] nums) {
        // 是否可以在数组中凑出sum / 2 -> 0-1背包
        // dp[j]: 是否能恰好装满j
        // 状态转移：dp[j] = dp[j] || dp[j - nums[i]]
        // dp[0] = true, 什么都不选，即可装满容量为0的背包
        int sum = Arrays.stream(nums).sum();
        if (sum % 2 == 1)
            return false;
        int target = sum / 2;
        boolean[] dp = new boolean[target + 1];
        dp[0] = true;
        for (int i = 0; i < nums.length; ++i) {
            for (int j = target; j >= nums[i]; --j) {
                dp[j] |= dp[j - nums[i]];
            }
        }
        return dp[target];
    }
}

329. 矩阵中的最长递增路径

BFS

class Solution {
    private int[][] dirs = new int[][]{{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
    int[][] path;
    public int longestIncreasingPath(int[][] matrix) {
        int m = matrix.length, n = matrix[0].length;
        path = new int[m][n];
        int res = 0;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                res = Math.max(res, dfs(i, j, m, n, matrix));
            }
        }
        return res;
    }

    private int dfs(int i, int j, int m, int n, int[][] matrix) {
        if (path[i][j] != 0)
            return path[i][j];
        int maxPath = 1;
        for (int[] dir : dirs) {
            int row = i + dir[0], col = j + dir[1];
            int path = 1;
            if (isValid(row, col, m, n) && matrix[row][col] > matrix[i][j]) {
                path += dfs(row, col, m, n, matrix);
            }
            maxPath = Math.max(maxPath, path);
        }
        path[i][j] = maxPath;
        return maxPath;
    }
    private boolean isValid(int row, int col, int m, int n) {
        return row >= 0 && col >= 0 && row < m && col < n;
    }
}

1208. 尽可能使字符串相等

滑动窗口

class Solution {
    public int equalSubstring(String s, String t, int maxCost) {
        int n = s.length();
        int[] diff = new int[n];
        for (int i = 0; i < n; ++i) {
            diff[i] = Math.abs(s.charAt(i) - t.charAt(i));
        }
        int res = 0, start = 0, sum = 0;
        for (int end = 0; end < n; ++end) {
            sum += diff[end];
            while (maxCost < sum) {
                sum -= diff[start++];
            }
            res = Math.max(res, end - start + 1);
        }
        return res;
    }
}
